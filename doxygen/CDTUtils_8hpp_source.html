108 template <typename T>
109 PtTriLocation::Enum locatePointTriangle(
110  const V2d<T>& p,
111  const V2d<T>& v1,
112  const V2d<T>& v2,
113  const V2d<T>& v3)
114 {
115  using namespace predicates::adaptive;
116  PtTriLocation::Enum result = PtTriLocation::Inside;
117  PtLineLocation::Enum edgeCheck = locatePointLine(p, v1, v2);
118  if(edgeCheck == PtLineLocation::Right)
119  return PtTriLocation::Outside;
120  if(edgeCheck == PtLineLocation::OnLine)
121  result = PtTriLocation::OnEdge1;
122  edgeCheck = locatePointLine(p, v2, v3);
123  if(edgeCheck == PtLineLocation::Right)
124  return PtTriLocation::Outside;
125  if(edgeCheck == PtLineLocation::OnLine)
126  result = PtTriLocation::OnEdge2;
127  edgeCheck = locatePointLine(p, v3, v1);
128  if(edgeCheck == PtLineLocation::Right)
129  return PtTriLocation::Outside;
130  if(edgeCheck == PtLineLocation::OnLine)
131  result = PtTriLocation::OnEdge3;
132  return result;
133 }
