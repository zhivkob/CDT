96 void generateGridTriangles(
97  OutputIt outFirst,
98  const std::size_t xres,
99  const std::size_t yres)
100 {
101  for(std::size_t iy = 0; iy < yres; ++iy)
102  {
103  for(std::size_t ix = 0; ix < xres; ++ix)
104  {
105  // 2___3 v3
106  // |\ | /\
107  // | \ | n3/ \n2
108  // |__\| /____\
109  // 0 1 v1 n1 v2
110  const std::size_t i = iy * xres + ix;
111  const std::size_t iv = iy * (xres + 1) + ix;
112  const VertInd vv[4] = {iv, iv + 1, iv + xres + 1, iv + xres + 2};
113  Triangle t;
114 
115  t.vertices = {vv[0], vv[1], vv[2]};
116  t.neighbors = {
117  iy ? 2 * i - xres * 2 + 1 : noNeighbor,
118  2 * i + 1,
119  ix ? 2 * i - 1 : noNeighbor};
120  *outFirst++ = t;
121 
122  t.vertices = {vv[1], vv[3], vv[2]};
123  t.neighbors = {
124  ix < xres - 1 ? 2 * i + 2 : noNeighbor,
125  iy < yres - 1 ? 2 * i + xres * 2 : noNeighbor,
126  2 * i};
127  *outFirst++ = t;
128  }
129  }
