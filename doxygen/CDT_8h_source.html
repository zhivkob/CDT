522 template <
523  typename TVertexIter,
524  typename TGetVertexCoordX,
525  typename TGetVertexCoordY>
526 void Triangulation<T, TNearPointLocator>::insertVertices(
527  const TVertexIter first,
528  const TVertexIter last,
529  TGetVertexCoordX getX,
530  TGetVertexCoordY getY)
531 {
532  if(vertices.empty())
533  {
534  addSuperTriangle(envelopBox<T>(first, last, getX, getY));
535  }
536 
537  const std::size_t nExistingVerts = vertices.size();
538 
539  vertices.reserve(nExistingVerts + std::distance(first, last));
540  for(TVertexIter it = first; it != last; ++it)
541  addNewVertex(V2d<T>::make(getX(*it), getY(*it)), TriIndVec());
542 
543  switch(m_vertexInsertionOrder)
544  {
545  case VertexInsertionOrder::AsProvided:
546  for(TVertexIter it = first; it != last; ++it)
547  insertVertex(VertInd(nExistingVerts + std::distance(first, it)));
548  break;
549  case VertexInsertionOrder::Randomized:
550  std::vector<VertInd> ii(std::distance(first, last));
551  typedef std::vector<VertInd>::iterator Iter;
552  VertInd value = nExistingVerts;
553  for(Iter it = ii.begin(); it != ii.end(); ++it, ++value)
554  *it = value;
555  std::random_shuffle(ii.begin(), ii.end(), randomCDT);
556  for(Iter it = ii.begin(); it != ii.end(); ++it)
557  insertVertex(*it);
558  break;
559  }
560 }

586 template <
587  typename T,
588  typename TVertexIter,
589  typename TGetVertexCoordX,
590  typename TGetVertexCoordY>
591 DuplicatesInfo FindDuplicates(
592  TVertexIter first,
593  TVertexIter last,
594  TGetVertexCoordX getX,
595  TGetVertexCoordY getY)
596 {
597  typedef unordered_map<V2d<T>, std::size_t> PosToIndex;
598  PosToIndex uniqueVerts;
599  const std::size_t verticesSize = std::distance(first, last);
600  DuplicatesInfo di = {
601  std::vector<std::size_t>(verticesSize), std::vector<std::size_t>()};
602  for(std::size_t iIn = 0, iOut = iIn; iIn < verticesSize; ++iIn, ++first)
603  {
604  typename PosToIndex::const_iterator it;
605  bool isUnique;
606  tie(it, isUnique) = uniqueVerts.insert(
607  std::make_pair(V2d<T>::make(getX(*first), getY(*first)), iOut));
608  if(isUnique)
609  {
610  di.mapping[iIn] = iOut++;
611  continue;
612  }
613  di.mapping[iIn] = it->second; // found a duplicate
614  di.duplicates.push_back(iIn);
615  }
616  return di;
617 }
