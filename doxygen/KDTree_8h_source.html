98  void
99  insert(const point_index& iPoint, const std::vector<point_type>& points)
100  {
101  // if point is outside root, extend tree by adding new roots
102  const point_type& pos = points[iPoint];
103  while(!isInsideBox(pos, m_min, m_max))
104  {
105  extendTree(pos);
106  }
107  // now insert the point into the tree
108  node_index node = m_root;
109  point_type min = m_min;
110  point_type max = m_max;
111  NodeSplitDirection::Enum dir = m_rootDir;
112 
113  coord_type mid;
114  NodeSplitDirection::Enum newDir;
115  point_type newMin, newMax;
116  while(true)
117  {
118  if(m_nodes[node].isLeaf())
119  {
120  // add point if capacity is not reached
121  point_data_vec& pd = m_nodes[node].data;
122  if(pd.size() < NumVerticesInLeaf)
123  {
124  pd.push_back(iPoint);
125  return;
126  }
127  // initialize bbox first time the root capacity is reached
128  if(!m_isRootBoxInitialized)
129  {
130  initializeRootBox(points);
131  min = m_min;
132  max = m_max;
133  }
134  // split a full leaf node
135  calcSplitInfo(min, max, dir, mid, newDir, newMin, newMax);
136  const node_index c1 = addNewNode(), c2 = addNewNode();
137  Node& n = m_nodes[node];
138  n.setChildren(c1, c2);
139  point_data_vec& c1data = m_nodes[c1].data;
140  point_data_vec& c2data = m_nodes[c2].data;
141  // move node's points to children
142  for(pd_cit it = n.data.begin(); it != n.data.end(); ++it)
143  {
144  whichChild(points[*it], mid, dir) == 0
145  ? c1data.push_back(*it)
146  : c2data.push_back(*it);
147  }
148  n.data = point_data_vec();
149  }
150  else
151  {
152  calcSplitInfo(min, max, dir, mid, newDir, newMin, newMax);
153  }
154  // add the point to a child
155  const std::size_t iChild = whichChild(points[iPoint], mid, dir);
156  iChild == 0 ? max = newMax : min = newMin;
157  node = m_nodes[node].children[iChild];
158  dir = newDir;
159  }

166  value_type nearest(
167  const point_type& point,
168  const std::vector<point_type>& points) const
169  {
170  value_type out;
171  int iTask = -1;
172  coord_type minDistSq = std::numeric_limits<coord_type>::max();
173  m_tasksStack[++iTask] =
174  NearestTask(m_root, m_min, m_max, m_rootDir, minDistSq);
175  while(iTask != -1)
176  {
177  const NearestTask t = m_tasksStack[iTask--];
178  if(t.distSq > minDistSq)
179  continue;
180  const Node& n = m_nodes[t.node];
181  if(n.isLeaf())
182  {
183  for(pd_cit it = n.data.begin(); it != n.data.end(); ++it)
184  {
185  const point_type& p = points[*it];
186  const coord_type distSq = CDT::distanceSquared(point, p);
187  if(distSq < minDistSq)
188  {
189  minDistSq = distSq;
190  out.first = p;
191  out.second = *it;
192  }
193  }
194  }
195  else
196  {
197  coord_type mid;
198  NodeSplitDirection::Enum newDir;
199  point_type newMin, newMax;
200  calcSplitInfo(t.min, t.max, t.dir, mid, newDir, newMin, newMax);
201 
202  const coord_type distToMid = t.dir == NodeSplitDirection::X
203  ? (point.x - mid)
204  : (point.y - mid);
205  const coord_type toMidSq = distToMid * distToMid;
206 
207  const std::size_t iChild = whichChild(point, mid, t.dir);
208  if(iTask + 2 >= m_tasksStack.size())
209  {
210  m_tasksStack.resize(
211  m_tasksStack.size() + StackDepthIncrement);
212  }
213  // node containing point should end up on top of the stack
214  if(iChild == 0)
215  {
216  m_tasksStack[++iTask] = NearestTask(
217  n.children[1], newMin, t.max, newDir, toMidSq);
218  m_tasksStack[++iTask] = NearestTask(
219  n.children[0], t.min, newMax, newDir, toMidSq);
220  }
221  else
222  {
223  m_tasksStack[++iTask] = NearestTask(
224  n.children[0], t.min, newMax, newDir, toMidSq);
225  m_tasksStack[++iTask] = NearestTask(
226  n.children[1], newMin, t.max, newDir, toMidSq);
227  }
228  }
229  }
230  return out;
231  }

293  void extendTree(const point_type& point)
294  {
295  const node_index newRoot = addNewNode();
296  const node_index newLeaf = addNewNode();
297  switch(m_rootDir)
298  {
299  case NodeSplitDirection::X:
300  m_rootDir = NodeSplitDirection::Y;
301  point.y < m_min.y ? m_nodes[newRoot].setChildren(newLeaf, m_root)
302  : m_nodes[newRoot].setChildren(m_root, newLeaf);
303  if(point.y < m_min.y)
304  m_min.y -= m_max.y - m_min.y;
305  else if(point.y > m_max.y)
306  m_max.y += m_max.y - m_min.y;
307  break;
308  case NodeSplitDirection::Y:
309  m_rootDir = NodeSplitDirection::X;
310  point.x < m_min.x ? m_nodes[newRoot].setChildren(newLeaf, m_root)
311  : m_nodes[newRoot].setChildren(m_root, newLeaf);
312  if(point.x < m_min.x)
313  m_min.x -= m_max.x - m_min.x;
314  else if(point.x > m_max.x)
315  m_max.x += m_max.x - m_min.x;
316  break;
317  }
318  m_root = newRoot;
319  }
