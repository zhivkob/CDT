161 template <
162  typename T,
163  typename TVertexIter,
164  typename TGetVertexCoordX,
165  typename TGetVertexCoordY>
166 CDT_EXPORT Box2d<T> envelopBox(
167  TVertexIter first,
168  TVertexIter last,
169  TGetVertexCoordX getX,
170  TGetVertexCoordY getY)
171 {
172  const T max = std::numeric_limits<T>::max();
173  Box2d<T> box = {{max, max}, {-max, -max}};
174  for(; first != last; ++first)
175  {
176  box.min.x = std::min(getX(*first), box.min.x);
177  box.max.x = std::max(getX(*first), box.max.x);
178  box.min.y = std::min(getY(*first), box.min.y);
179  box.max.y = std::max(getY(*first), box.max.y);
180  }
181  return box;
182 }

406 struct hash<CDT::Edge>
407 {
409  std::size_t operator()(const CDT::Edge& e) const
410  {
411  return hashEdge(e);
412  }
413 
414 private:
415  static void hashCombine(std::size_t& seed, const CDT::VertInd& key)
416  {
417 #ifdef CDT_CXX11_IS_SUPPORTED
418  typedef std::hash<CDT::VertInd> Hasher;
419 #else
420  typedef boost::hash<CDT::VertInd> Hasher;
421 #endif
422  seed ^= Hasher()(key) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
423  }
424  static std::size_t hashEdge(const CDT::Edge& e)
425  {
426  const std::pair<CDT::VertInd, CDT::VertInd>& vv = e.verts();
427  std::size_t seed1(0);
428  hashCombine(seed1, vv.first);
429  hashCombine(seed1, vv.second);
430  std::size_t seed2(0);
431  hashCombine(seed2, vv.second);
432  hashCombine(seed2, vv.first);
433  return std::min(seed1, seed2);
434  }
435 };
