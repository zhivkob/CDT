72 template <typename T, typename TNearPointLocator>
73 void Triangulation<T, TNearPointLocator>::eraseDummies()
74 {
75  if(m_dummyTris.empty())
76  return;
77  const TriIndUSet dummySet(m_dummyTris.begin(), m_dummyTris.end());
78  TriIndUMap triIndMap;
79  triIndMap[noNeighbor] = noNeighbor;
80  for(TriInd iT(0), iTnew(0); iT < TriInd(triangles.size()); ++iT)
81  {
82  if(dummySet.count(iT))
83  continue;
84  triIndMap[iT] = iTnew;
85  triangles[iTnew] = triangles[iT];
86  iTnew++;
87  }
88  triangles.erase(triangles.end() - dummySet.size(), triangles.end());
89 
90  // remap adjacent triangle indices for vertices
91  typedef typename VerticesTriangles::iterator VertTrisIt;
92  for(VertTrisIt vTris = vertTris.begin(); vTris != vertTris.end(); ++vTris)
93  {
94  for(TriIndVec::iterator iT = vTris->begin(); iT != vTris->end(); ++iT)
95  *iT = triIndMap[*iT];
96  }
97  // remap neighbor indices for triangles
98  for(TriangleVec::iterator t = triangles.begin(); t != triangles.end(); ++t)
99  {
100  NeighborsArr3& nn = t->neighbors;
101  for(NeighborsArr3::iterator iN = nn.begin(); iN != nn.end(); ++iN)
102  *iN = triIndMap[*iN];
103  }
104  // clear dummy triangles
105  m_dummyTris = std::vector<TriInd>();
106 }

282 template <typename T, typename TNearPointLocator>
283 void Triangulation<T, TNearPointLocator>::insertEdge(Edge edge)
284 {
285  const VertInd iA = edge.v1();
286  VertInd iB = edge.v2();
287  if(iA == iB) // edge connects a vertex to itself
288  return;
289  const TriIndVec& aTris = vertTris[iA];
290  const TriIndVec& bTris = vertTris[iB];
291  const V2d<T>& a = vertices[iA];
292  const V2d<T>& b = vertices[iB];
293  if(verticesShareEdge(aTris, bTris))
294  {
295  fixEdge(Edge(iA, iB));
296  return;
297  }
298  TriInd iT;
299  VertInd iVleft, iVright;
300  tie(iT, iVleft, iVright) = intersectedTriangle(iA, aTris, a, b);
301  // if one of the triangle vertices is on the edge, move edge start
302  if(iT == noNeighbor)
303  {
304  fixEdge(Edge(iA, iVleft));
305  return insertEdge(Edge(iVleft, iB));
306  }
307  std::vector<TriInd> intersected(1, iT);
308  std::vector<VertInd> ptsLeft(1, iVleft);
309  std::vector<VertInd> ptsRight(1, iVright);
310  VertInd iV = iA;
311  Triangle t = triangles[iT];
312  const VerticesArr3& tverts = t.vertices;
313  while(std::find(tverts.begin(), tverts.end(), iB) == tverts.end())
314  {
315  const TriInd iTopo = opposedTriangle(t, iV);
316  const Triangle& tOpo = triangles[iTopo];
317  const VertInd iVopo = opposedVertex(tOpo, iT);
318  const V2d<T> vOpo = vertices[iVopo];
319 
320  intersected.push_back(iTopo);
321  iT = iTopo;
322  t = triangles[iT];
323 
324  const PtLineLocation::Enum loc = locatePointLine(vOpo, a, b);
325  if(loc == PtLineLocation::Left)
326  {
327  ptsLeft.push_back(iVopo);
328  iV = iVleft;
329  iVleft = iVopo;
330  }
331  else if(loc == PtLineLocation::Right)
332  {
333  ptsRight.push_back(iVopo);
334  iV = iVright;
335  iVright = iVopo;
336  }
337  else // encountered point on the edge
338  iB = iVopo;
339  }
340  // Remove intersected triangles
341  typedef std::vector<TriInd>::const_iterator TriIndCit;
342  for(TriIndCit it = intersected.begin(); it != intersected.end(); ++it)
343  makeDummy(*it);
344  // Triangulate pseudo-polygons on both sides
345  const TriInd iTleft = triangulatePseudopolygon(iA, iB, ptsLeft);
346  std::reverse(ptsRight.begin(), ptsRight.end());
347  const TriInd iTright = triangulatePseudopolygon(iB, iA, ptsRight);
348  changeNeighbor(iTleft, noNeighbor, iTright);
349  changeNeighbor(iTright, noNeighbor, iTleft);
350  // add fixed edge
351  fixEdge(Edge(iA, iB));
352  if(iB != edge.v2()) // encountered point on the edge
353  return insertEdge(Edge(iB, edge.v2()));
354 }

366 template <typename T, typename TNearPointLocator>
367 tuple<TriInd, VertInd, VertInd>
368 Triangulation<T, TNearPointLocator>::intersectedTriangle(
369  const VertInd iA,
370  const std::vector<TriInd>& candidates,
371  const V2d<T>& a,
372  const V2d<T>& b) const
373 {
374  typedef std::vector<TriInd>::const_iterator TriIndCit;
375  for(TriIndCit it = candidates.begin(); it != candidates.end(); ++it)
376  {
377  const TriInd iT = *it;
378  const Triangle t = triangles[iT];
379  const Index i = vertexInd(t, iA);
380  const VertInd iP1 = t.vertices[cw(i)];
381  const VertInd iP2 = t.vertices[ccw(i)];
382  const PtLineLocation::Enum locP1 = locatePointLine(vertices[iP1], a, b);
383  const PtLineLocation::Enum locP2 = locatePointLine(vertices[iP2], a, b);
384  if(locP2 == PtLineLocation::Right)
385  {
386  if(locP1 == PtLineLocation::OnLine)
387  return make_tuple(noNeighbor, iP1, iP2);
388  if(locP1 == PtLineLocation::Left)
389  return make_tuple(iT, iP1, iP2);
390  }
391  }
392  throw std::runtime_error("Could not find vertex triangle intersected by "
393  "edge. Note: can be caused by duplicate points.");
394 }

396 template <typename T, typename TNearPointLocator>
397 void Triangulation<T, TNearPointLocator>::addSuperTriangle(const Box2d<T>& box)
398 {
399  m_nTargetVerts = 3;
400  m_superGeomType = SuperGeometryType::SuperTriangle;
401 
402  const V2d<T> center = {
403  (box.min.x + box.max.x) / T(2), (box.min.y + box.max.y) / T(2)};
404  const T w = box.max.x - box.min.x;
405  const T h = box.max.y - box.min.y;
406  T r = std::sqrt(w * w + h * h) / T(2); // incircle radius
407  r *= T(1.1);
408  const T R = T(2) * r; // excircle radius
409  const T shiftX = R * std::sqrt(T(3)) / T(2); // R * cos(30 deg)
410  const V2d<T> posV1 = {center.x - shiftX, center.y - r};
411  const V2d<T> posV2 = {center.x + shiftX, center.y - r};
412  const V2d<T> posV3 = {center.x, center.y + R};
413  addNewVertex(posV1, TriIndVec(1, TriInd(0)));
414  addNewVertex(posV2, TriIndVec(1, TriInd(0)));
415  addNewVertex(posV3, TriIndVec(1, TriInd(0)));
416  const Triangle superTri = {
417  {VertInd(0), VertInd(1), VertInd(2)},
418  {noNeighbor, noNeighbor, noNeighbor}};
419  addTriangle(superTri);
420 
421  m_nearPtLocator.addPoint(VertInd(0), vertices);
422  m_nearPtLocator.addPoint(VertInd(1), vertices);
423  m_nearPtLocator.addPoint(VertInd(2), vertices);
424 }

435 template <typename T, typename TNearPointLocator>
436 void Triangulation<T, TNearPointLocator>::insertVertex(const VertInd iVert)
437 {
438  const V2d<T>& v = vertices[iVert];
439  array<TriInd, 2> trisAt = walkingSearchTrianglesAt(v);
440  std::stack<TriInd> triStack =
441  trisAt[1] == noNeighbor
442  ? insertPointInTriangle(iVert, trisAt[0])
443  : insertPointOnEdge(iVert, trisAt[0], trisAt[1]);
444  while(!triStack.empty())
445  {
446  const TriInd iT = triStack.top();
447  triStack.pop();
448 
449  const Triangle& t = triangles[iT];
450  const TriInd iTopo = opposedTriangle(t, iVert);
451  if(iTopo == noNeighbor)
452  continue;
453  if(isFlipNeeded(v, iT, iTopo, iVert))
454  {
455  flipEdge(iT, iTopo);
456  triStack.push(iT);
457  triStack.push(iTopo);
458  }
459  }
460 
461  m_nearPtLocator.addPoint(iVert, vertices);
462 }

474 template <typename T, typename TNearPointLocator>
475 bool Triangulation<T, TNearPointLocator>::isFlipNeeded(
476  const V2d<T>& pos,
477  const TriInd iT,
478  const TriInd iTopo,
479  const VertInd iVert) const
480 {
481  const Triangle& tOpo = triangles[iTopo];
482  const Index i = opposedVertexInd(tOpo, iT);
483  const VertInd iVopo = tOpo.vertices[i];
484  const VertInd iVcw = tOpo.vertices[cw(i)];
485  const VertInd iVccw = tOpo.vertices[ccw(i)];
486  const V2d<T>& v1 = vertices[iVcw];
487  const V2d<T>& v2 = vertices[iVopo];
488  const V2d<T>& v3 = vertices[iVccw];
489  if(m_superGeomType == SuperGeometryType::SuperTriangle)
490  {
491  if(iVert < 3 || iVopo < 3) // flip-candidate edge touches super-triangle
492  {
493  if(iVcw < 3 || iVccw < 3) // but so does original edge
494  {
495  // let the normal circumcircle test decide
496  return isInCircumcircle(pos, v1, v2, v3);
497  }
498  return false; // no flip is needed
499  }
500 
501  if(iVcw < 3)
502  return locatePointLine(v1, v2, v3) == locatePointLine(pos, v2, v3);
503  if(iVccw < 3)
504  return locatePointLine(v3, v1, v2) == locatePointLine(pos, v1, v2);
505  }
506  return isInCircumcircle(pos, v1, v2, v3);
507 }

509 /* Insert point into triangle: split into 3 triangles:
510  * - create 2 new triangles
511  * - re-use old triangle for the 3rd
512  * v3
513  * / | \
514  * / | \ <-- original triangle (t)
515  * / | \
516  * n3 / | \ n2
517  * /newT2|newT1\
518  * / v \
519  * / __/ \__ \
520  * / __/ \__ \
521  * / _/ t' \_ \
522  * v1 ___________________ v2
523  * n1
524  */
525 template <typename T, typename TNearPointLocator>
526 std::stack<TriInd> Triangulation<T, TNearPointLocator>::insertPointInTriangle(
527  const VertInd v,
528  const TriInd iT)
529 {
530  const TriInd iNewT1 = addTriangle();
531  const TriInd iNewT2 = addTriangle();
532 
533  Triangle& t = triangles[iT];
534  const array<VertInd, 3> vv = t.vertices;
535  const array<TriInd, 3> nn = t.neighbors;
536  const VertInd v1 = vv[0], v2 = vv[1], v3 = vv[2];
537  const TriInd n1 = nn[0], n2 = nn[1], n3 = nn[2];
538  // make two new triangles and convert current triangle to 3rd new triangle
539  using detail::arr3;
540  triangles[iNewT1] = Triangle::make(arr3(v2, v3, v), arr3(n2, iNewT2, iT));
541  triangles[iNewT2] = Triangle::make(arr3(v3, v1, v), arr3(n3, iT, iNewT1));
542  t = Triangle::make(arr3(v1, v2, v), arr3(n1, iNewT1, iNewT2));
543  // make and add a new vertex
544  addAdjacentTriangles(v, iT, iNewT1, iNewT2);
545  // adjust lists of adjacent triangles for v1, v2, v3
546  addAdjacentTriangle(v1, iNewT2);
547  addAdjacentTriangle(v2, iNewT1);
548  removeAdjacentTriangle(v3, iT);
549  addAdjacentTriangle(v3, iNewT1);
550  addAdjacentTriangle(v3, iNewT2);
551  // change triangle neighbor's neighbors to new triangles
552  changeNeighbor(n2, iT, iNewT1);
553  changeNeighbor(n3, iT, iNewT2);
554  // return newly added triangles
555  std::stack<TriInd> newTriangles;
556  newTriangles.push(iT);
557  newTriangles.push(iNewT1);
558  newTriangles.push(iNewT2);
559  return newTriangles;
560 }

562 /* Inserting a point on the edge between two triangles
563  * T1 (top) v1
564  * /|\
565  * n1 / | \ n4
566  * / | \
567  * / T1' | Tnew1\
568  * v2-------v-------v4
569  * \ Tnew2| T2' /
570  * \ | /
571  * n2 \ | / n3
572  * \|/
573  * T2 (bottom) v3
574  */
575 template <typename T, typename TNearPointLocator>
576 std::stack<TriInd> Triangulation<T, TNearPointLocator>::insertPointOnEdge(
577  const VertInd v,
578  const TriInd iT1,
579  const TriInd iT2)
580 {
581  const TriInd iTnew1 = addTriangle();
582  const TriInd iTnew2 = addTriangle();
583 
584  Triangle& t1 = triangles[iT1];
585  Triangle& t2 = triangles[iT2];
586  Index i = opposedVertexInd(t1, iT2);
587  const VertInd v1 = t1.vertices[i];
588  const VertInd v2 = t1.vertices[ccw(i)];
589  const TriInd n1 = t1.neighbors[i];
590  const TriInd n4 = t1.neighbors[cw(i)];
591  i = opposedVertexInd(t2, iT1);
592  const VertInd v3 = t2.vertices[i];
593  const VertInd v4 = t2.vertices[ccw(i)];
594  const TriInd n3 = t2.neighbors[i];
595  const TriInd n2 = t2.neighbors[cw(i)];
596  // add new triangles and change existing ones
597  using detail::arr3;
598  t1 = Triangle::make(arr3(v1, v2, v), arr3(n1, iTnew2, iTnew1));
599  t2 = Triangle::make(arr3(v3, v4, v), arr3(n3, iTnew1, iTnew2));
600  triangles[iTnew1] = Triangle::make(arr3(v1, v, v4), arr3(iT1, iT2, n4));
601  triangles[iTnew2] = Triangle::make(arr3(v3, v, v2), arr3(iT2, iT1, n2));
602  // make and add new vertex
603  addAdjacentTriangles(v, iT1, iTnew2, iT2, iTnew1);
604  // adjust neighboring triangles and vertices

644 template <typename T, typename TNearPointLocator>
645 TriInd Triangulation<T, TNearPointLocator>::walkTriangles(
646  const VertInd startVertex,
647  const V2d<T>& pos) const
648 {
649  // begin walk in search of triangle at pos
650  TriInd currTri = vertTris[startVertex][0];
651 #ifdef CDT_USE_BOOST
652  TriIndFlatUSet visited;

662  const Index offset(detail::randGen() % 3);
663  for(Index i_(0); i_ < Index(3); ++i_)
664  {
665  const Index i((i_ + offset) % 3);

671  t.neighbors[i] != noNeighbor &&
672  visited.insert(t.neighbors[i]).second)
673  {
674  found = false;
675  currTri = t.neighbors[i];

678  }

705 /* Flip edge between T and Topo:
706  *
707  * v4 | - old edge
708  * /|\ ~ - new edge
709  * / | \
710  * n3 / T' \ n4
711  * / | \
712  * / | \
713  * T -> v1~~~~~~~~~v3 <- Topo
714  * \ | /
715  * \ | /
716  * n1 \Topo'/ n2
717  * \ | /
718  * \|/
719  * v2
720  */
721 template <typename T, typename TNearPointLocator>
722 void Triangulation<T, TNearPointLocator>::flipEdge(
723  const TriInd iT,
724  const TriInd iTopo)
725 {
726  Triangle& t = triangles[iT];
727  Triangle& tOpo = triangles[iTopo];
728  const array<TriInd, 3>& triNs = t.neighbors;
729  const array<TriInd, 3>& triOpoNs = tOpo.neighbors;
730  const array<VertInd, 3>& triVs = t.vertices;
731  const array<VertInd, 3>& triOpoVs = tOpo.vertices;
732  // find vertices and neighbors
733  Index i = opposedVertexInd(t, iTopo);
734  const VertInd v1 = triVs[i];
735  const VertInd v2 = triVs[ccw(i)];
736  const TriInd n1 = triNs[i];
737  const TriInd n3 = triNs[cw(i)];
738  i = opposedVertexInd(tOpo, iT);
739  const VertInd v3 = triOpoVs[i];
740  const VertInd v4 = triOpoVs[ccw(i)];
741  const TriInd n4 = triOpoNs[i];
742  const TriInd n2 = triOpoNs[cw(i)];
743  // change vertices and neighbors
744  using detail::arr3;
745  t = Triangle::make(arr3(v4, v1, v3), arr3(n3, iTopo, n4));
746  tOpo = Triangle::make(arr3(v2, v3, v1), arr3(n2, iT, n1));
747  // adjust neighboring triangles and vertices
748  changeNeighbor(n1, iT, iTopo);
749  changeNeighbor(n4, iTopo, iT);
750  addAdjacentTriangle(v1, iTopo);
751  addAdjacentTriangle(v3, iT);
752  removeAdjacentTriangle(v2, iT);
753  removeAdjacentTriangle(v4, iTopo);
754 }

828 template <typename T, typename TNearPointLocator>
829 TriInd Triangulation<T, TNearPointLocator>::triangulatePseudopolygon(
830  const VertInd ia,
831  const VertInd ib,
832  const std::vector<VertInd>& points)
833 {
834  if(points.empty())
835  return pseudopolyOuterTriangle(ia, ib);
836  const VertInd ic = findDelaunayPoint(ia, ib, points);
837  const std::pair<std::vector<VertInd>, std::vector<VertInd> > splitted =
838  splitPseudopolygon(ic, points);
839  // triangulate splitted pseudo-polygons
840  TriInd iT2 = triangulatePseudopolygon(ic, ib, splitted.second);
841  TriInd iT1 = triangulatePseudopolygon(ia, ic, splitted.first);
842  // add new triangle
843  const Triangle t = {{ia, ib, ic}, {noNeighbor, iT2, iT1}};
844  const TriInd iT = addTriangle(t);
845  // adjust neighboring triangles and vertices
846  if(iT1 != noNeighbor)
847  {
848  if(splitted.first.empty())
849  changeNeighbor(iT1, ia, ic, iT);
850  else
851  triangles[iT1].neighbors[0] = iT;
852  }
853  if(iT2 != noNeighbor)
854  {
855  if(splitted.second.empty())
856  changeNeighbor(iT2, ic, ib, iT);
857  else
858  triangles[iT2].neighbors[0] = iT;
859  }
860  addAdjacentTriangle(ia, iT);
861  addAdjacentTriangle(ib, iT);
862  addAdjacentTriangle(ic, iT);
863 
864  return iT;
865 }

939 CDT_INLINE_IF_HEADER_ONLY
940 unordered_map<TriInd, LayerDepth> PeelLayer(
941  std::stack<TriInd> seeds,
942  const TriangleVec& triangles,
943  const EdgeUSet& fixedEdges,
944  const unordered_map<Edge, BoundaryOverlapCount>& overlapCount,
945  const LayerDepth layerDepth,
946  std::vector<LayerDepth>& triDepths)
947 {
948  unordered_map<TriInd, LayerDepth> behindBoundary;
949  while(!seeds.empty())
950  {
951  const TriInd iT = seeds.top();
952  seeds.pop();
953  triDepths[iT] = layerDepth;
954  behindBoundary.erase(iT);
955  const Triangle& t = triangles[iT];
956  for(Index i(0); i < Index(3); ++i)
957  {
958  const Edge opEdge(t.vertices[ccw(i)], t.vertices[cw(i)]);
959  const TriInd iN = t.neighbors[opoNbr(i)];
960  if(iN == noNeighbor || triDepths[iN] <= layerDepth)
961  continue;
962  if(fixedEdges.count(opEdge))
963  {
964  const unordered_map<Edge, LayerDepth>::const_iterator cit =
965  overlapCount.find(opEdge);
966  const LayerDepth triDepth = cit == overlapCount.end()
967  ? layerDepth + 1
968  : layerDepth + cit->second + 1;
969  behindBoundary[iN] = triDepth;
970  continue;
971  }
972  seeds.push(iN);
973  }
974  }
975  return behindBoundary;
976 }

1012  const TriInd seed,
1013  const TriangleVec& triangles,
1014  const EdgeUSet& fixedEdges,
1015  const unordered_map<Edge, BoundaryOverlapCount>& overlapCount)
1016 {
1017  std::vector<LayerDepth> triDepths(
1018  triangles.size(), std::numeric_limits<LayerDepth>::max());
1019  std::stack<TriInd> seeds(TriDeque(1, seed));
1020  LayerDepth layerDepth = 0;
1021  LayerDepth deepestSeedDepth = 0;
1022 
1023  unordered_map<LayerDepth, TriIndUSet> seedsByDepth;
1024  do
1025  {

1027  seeds, triangles, fixedEdges, overlapCount, layerDepth, triDepths);
1028 
1029  seedsByDepth.erase(layerDepth);
1030  typedef unordered_map<TriInd, LayerDepth>::const_iterator Iter;
1031  for(Iter it = newSeeds.begin(); it != newSeeds.end(); ++it)
1032  {

1034  seedsByDepth[it->second].insert(it->first);

1037  seeds = std::stack<TriInd>(
1038  TriDeque(nextLayerSeeds.begin(), nextLayerSeeds.end()));
1039  ++layerDepth;
1040  } while(!seeds.empty() || deepestSeedDepth > layerDepth);
1041 
1042  return triDepths;
1043 }
